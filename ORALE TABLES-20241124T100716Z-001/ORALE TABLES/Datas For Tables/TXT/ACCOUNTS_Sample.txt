CREATE TABLE  "ACCOUNTS" 
   (	"NAME" VARCHAR2(64) NOT NULL ENABLE, 
	"CONTACT" VARCHAR2(10) NOT NULL ENABLE, 
	"ADDRESS" VARCHAR2(128) NOT NULL ENABLE, 
	"EMAIL" VARCHAR2(128) NOT NULL ENABLE, 
	"USERNAME" VARCHAR2(128) NOT NULL ENABLE, 
	"PASSWORD" VARCHAR2(256) NOT NULL ENABLE, 
	"ROLE" NUMBER NOT NULL ENABLE, 
	"VERIFYKEY" VARCHAR2(128) NOT NULL ENABLE, 
	"VERIFYCHECK" NUMBER NOT NULL ENABLE, 
	"REGISTERTIME" VARCHAR2(256) NOT NULL ENABLE, 
	"ACCOUNTID" NUMBER NOT NULL ENABLE, 
	"CATEGORY" VARCHAR2(32), 
	 PRIMARY KEY ("ACCOUNTID") ENABLE
   )
/

CREATE OR REPLACE TRIGGER  "ACCOUNTID_AITRIGGER" BEFORE INSERT ON accounts  FOR EACH ROW   WHEN (new.AccountID IS NULL) begin 
:new.AccountID := seq_accountid.nextVal; 
END; 

/
ALTER TRIGGER  "ACCOUNTID_AITRIGGER" ENABLE
/
